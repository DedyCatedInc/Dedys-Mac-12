nano ~/.zshrc

export NVM_DIR="$HOME/.nvm"

[ -s "/usr/local/opt/nvm/etc/bash_completion.d/nvm" ] && . "/usr/local/opt/nvm/etc/bash_completion.d/nvm"  # This loads nvm bash_completion

Then exit terminal and reopen again
